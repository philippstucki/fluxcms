  81.     function showPage ($configXml,$PageOptions = array(), $returnDb2XmlObject = false) 
    82. 

  82.     {
    83. 

  83.  
    84. 

  84.         
    85. 

  85.           require_once("bitlib/admin/php/api.php");
    86. 

  86.         $this->api = admin_api::getInstance();
    87. 

  87.         // get the queries, either cached from file system or generated
    88. 

  88.  
    89. 

  89.         if (is_null($this->queries)) {
    90. 

  90.             $this->queries = $this->getQueries($configXml,$PageOptions);
    91. 

  91.         }
    92. 

  92.  
    93. 

  93.         if (phpversion("domxml") >= 20030405) {
    94. 

  94.             $sql2xml = new xml_db2xml($this->db,"iba","Extended_NG",array("Sql_NG","Dbresult_NG","Array","Http","File","XmlObject","String"));
    95. 

  95.         } else {
    96. 

  96.             $sql2xml = new xml_db2xml($this->db,"iba","Extended");
    97. 

  97.         }
    98. 

  98.  
    99. 

  99.         // i should add this for all options .... later maybe
    100. 

  100.         if (!(is_null($this->getAttrib("xml_seperator")) ))
    101. 

  101.         {
    102. 

  102.             $sql2xml->setOptions(array("user_options" => array("xml_seperator"=>$this->getAttrib("xml_seperator"))));
    103. 

  103.         }
    104. 

  104.         /*I'm not sure, if we need the structure.xml in the output....  normaly we don't */
    105. 

  105.         if (isset($PageOptions["All"]["include_structure_xml"]) && $PageOptions["All"]["include_structure_xml"])
    106. 

  106.         {
    107. 

  107.             $sql2xml->add($configXml);
    108. 

  108.         }
    109. 

  109.         
    110. 

  110.         $sql2xml->setOptions(array("user_options" => array("print_empty_ids"=>False)));
    111. 

  111.  
    112. 

  112.         
    113. 

  113.         
    114. 

  114.         if (isset($PageOptions["All"]["user_options"])){
    115. 

  115.             $sql2xml->setOptions(array("user_options" => $PageOptions["All"]["user_options"]));
    116. 

  116.         }
    117. 

  117.  
    118. 

  118.         if (isset($PageOptions["querystring"])){
    119. 

  119.             $sql2xml->setOptions(array("user_options"=>array("result_root"=>"querystring")));
    120. 

  120.             $pageo = array();
    121. 

  121.             foreach($PageOptions["querystring"] as $key => $value)
    122. 

  122.             {
    123. 

  123.                 if (!is_array($value))
    124. 

  124.                 {
    125. 

  125.                     $pageo[$key] = $value;
    126. 

  126.                 }
    127. 

  127.             }
    128. 

  128.             $sql2xml->add($pageo);
    129. 

  129.         }
    130. 

  130.         if ($array2xml = $this->getParameter("array2xml")) {
    131. 

  131.             foreach ($array2xml as $name => $array) {
    132. 

  132.                 $sql2xml->setOptions(array("user_options"=>array("result_root"=>$name)));
    133. 

  133.                 $sql2xml->add($array);
    134. 

  134.             }
    135. 

  135.         }
    136. 

  136.         
    137. 

  137.     if (is_array($this->queries))
    138. 

  138.     {
    139. 

  139.         $_maxLastChangedAll = array(0);
    140. 

  140.  
    141. 

  141.         foreach ($this->queries as $structureName => $query) {
    142. 

  142.             if  ($structureName == "_queryInfo") {
    143. 

  143.                 continue;
    144. 

  144.             }
    145. 

  145.             $query['user_options']['result_root'] = $structureName;
    146. 

  146.             if ($query['type'] == "dbquery"){
    147. 

  147.                 //caching the sql2xml part
    148. 

  148.                 $query["query"] = $this->replaceVarsInWhere($query["query"]);
    149. 

  149.                 if ( $this->st2xmlCaching == "true" ) { 
    150. 

  150.                     if (! (isset($query["maxLastChanged"]) )) {
    151. 

  151.                         $query["maxLastChanged"]  = $this->db->getOne($query['queryLastChanged']);
    152. 

  152.                         
    153. 

  153.                      } 
    154. 

  154.                         
    155. 

  155.                     if ( $cachedXML = $this->api->simpleCacheCheck("","st2xml_data",$query['query'],"file", $query["maxLastChanged"])) {
    156. 

  156.                         $sql2xml->addWithInput("File",$cachedXML);
    157. 

  157.                     } 
    158. 

  158.                     else {
    159. 

  159.                         $sql2xml->setOptions(array("user_tableInfo"=>$query['tableInfo'],"user_options"=>$query['user_options']));
    160. 

  160.                         $sql2xml->add($query['query']);
    161. 

  161.                         $ctx = xpath_new_context($sql2xml->Format->xmldoc);
    162. 

  162.                         $resultTree = $ctx->xpath_eval("$structureName",$sql2xml->Format->xmlroot );
    163. 

  163.                         $this->api->simpleCacheWrite("","st2xml_data",$query['query'],"<?xml version='1.0' ?".">".$sql2xml->Format->xmldoc->dump_node($resultTree->nodeset[0]),"file", $query["maxLastChanged"]);
    164. 

  164.                     }
    165. 

  165.                     $_maxLastChangedAll[] = $query["maxLastChanged"];
    166. 

  166.                     
    167. 

  167.                 }
    168. 

  168.                 else
    169. 

  169.                 {
    170. 

  170.                         $sql2xml->setOptions(array("user_tableInfo"=>$query['tableInfo'],"user_options"=>$query['user_options']));
    171. 

  171.                         $sql2xml->add($query['query']);
    172. 

  172.                 }
    173. 

  173.             }
    174. 

  174.             else if ( $query['type'] == "aggregate" ) {
    175. 

  175.                 if ($this->st2xmlCaching == "true" ) {
    176. 

  176.                     //http stuff
    177. 

  177.                     if (strpos($query['query'],"http") === 0 ) {
    178. 

  178.  
    179. 

  179.                         //if maxLastChanged is set, we already checked it in the Validity container..
    180. 

  180.                         if (! (isset($query["maxLastChanged"]) )) {
    181. 

  181.                             //check for Last Modified automatically creates the cached file, if not already there
    182. 

  182.                             $query["maxLastChanged"] = $this->api->simpleCacheHttpLastModified($query['query'], time() - $query['expires'],$this->getParameter ("default","proxy") );
    183. 

  183.                         }
    184. 

  184.                         $_maxLastChangedAll[] = $query["maxLastChanged"];
    185. 

  185.                         $sql2xml->addWithInput("File",$this->api->simpleCacheGenerateName("simpleCacheHttp",$query['query']),array("root"=> $structureName));
    186. 

  186.                     } 
    187. 

  187.                     // if not starting with http, it will be a local file... hopefully
    188. 

  188.                     else {
    189. 

  189.                         $sql2xml->addWithInput("File",$query['query']);
    190. 

  190.                     }
    191. 

  191.                 } 
    192. 

  192.                 else {
    193. 

  193.                 // with no caching, just get and add it.. (maybe we should allow http caching without db-caching..)
    194. 

  194.                     $sql2xml->add($query['query'],array("root"=> $structureName));
    195. 

  195.                 }
    196. 

  196.             } 
    197. 

  197.             else {
    198. 

  198.             //quite useless right now... 
    199. 

  199.                 $sql2xml->add($query['query'],array("root"=> $structureName));
    200. 

  200.             }
    201. 

  201.         }
    202. 

  202.         $this->queries["_queryInfo"]["maxLastChanged"] = max ($_maxLastChangedAll);
    203. 

  203.         
    204. 

  204.         // send http cache headers so the sent page expires immediately 
    205. 

  205.         $lastChangedDate = gmdate("r", $this->queries["_queryInfo"]["maxLastChanged"]);
    206. 

  206.         $this->sitemap->setHeader("Last-Modified", $lastChangedDate);
    207. 

  207.         $this->sitemap->setHeader('Expires', $lastChangedDate);
    208. 

  208.         
    209. 

  209.         //var_dump(gmdate("r", $this->queries["_queryInfo"]["maxLastChanged"] ));
    210. 

  210.         // FIXME: if there is no bitlib2 code anymore, this if can go away
    211. 

  211.         if (method_exists($this->sitemap,"setLastModified")) {
    212. 

  212.             $this->sitemap->setLastModified($this->queries["_queryInfo"]["maxLastChanged"] );
    213. 

  213.         }
    214. 

  214.     }
    215. 

  215.     if ($returnDb2XmlObject) {
    216. 

  216.        return $sql2xml;   
    217. 

  217.     } else {
    218. 

  218.         return $sql2xml->getXmlObject();
    219. 

  219.     }
    220. 

  220.  
    221. 

  221.     }

  224.     function Structures2Sql ($configFile,$sqlOptions=array(),$rootpath= "/structure")
    225. 

  225.     {
    226. 

  226.  
    227. 

  227.         $configClass = common::getConfigClass($configFile);
    228. 

  228.  
    229. 

  229.         $dbMainStructure = $configClass->getValues( $rootpath);
    230. 

  230.         if (is_array($dbMainStructure['children']))
    231. 

  231.         {
    232. 

  232.             foreach ($dbMainStructure['children'] as $structureName) {
    233. 

  233.  
    234. 

  234.             $dbStructure = $configClass->getValues( "$rootpath/$structureName");
    235. 

  235.             if (isset($dbStructure["expires"])) {
    236. 

  236.                 $allqueries[$structureName]['expires'] = time() - strtotime(preg_replace("#access\s+minus\s+#", "-",$dbStructure["expires"]));
    237. 

  237.             } 
    238. 

  238.             else {
    239. 

  239.                 $allqueries[$structureName]['expires'] = $this->defaultExpires;
    240. 

  240.             }
    241. 

  241.             if (isset($dbStructure["type"]) && $dbStructure["type"] == "aggregate") {
    242. 

  242.  
    243. 

  243.                  $allqueries[$structureName]['query'] = $dbStructure["src"];
    244. 

  244.                 $allqueries[$structureName]['type'] = "aggregate";
    245. 

  245.             }
    246. 

  246.  
    247. 

  247.             else {
    248. 

  248. /*  This code looks like unnecessary...
    249. 

  249.     can be removed later, if it's really not needed 
    250. 

  250.  
    251. 

  251.             	 $queryfields = $dbStructure['children'][0].".*";
    252. 

  252.             	 $query = "from ".$dbStructure['children'][0];
    253. 

  253.             	 $name = $dbStructure['children'][0];;
    254. 

  254.             	 $dbStructure = $configClass->getValues("$rootpath/$structureName/$name");
    255. 

  255. */                 
    256. 

  256.                  $tableInfo= array();
    257. 

  257.                  if (!isset($sqlOptions[$structureName])) { $sqlOptions[$structureName] = array();} //E_ALL fix
    258. 

  258.  
    259. 

  259.                  $allqueries[$structureName]['query'] = $this->Structure2Sql($configClass,$tableInfo,$sqlOptions[$structureName],$rootpath."/".$structureName);
    260. 

  260.                  $allqueries[$structureName]['tableInfo'] = $tableInfo;
    261. 

  261.                  $this->printDebug("SQL-Query for Section: $structureName: ".$allqueries[$structureName]['query']);
    262. 

  262.                  $allqueries[$structureName]['type'] = "dbquery";
    263. 

  263.             }
    264. 

  264.         }
    265. 

  265.     }
    266. 

  266.    // print "<pre>";print_r($allqueries);
    267. 

  267.         return $allqueries;
    268. 

  268.  
    269. 

  269.     }

  271.     function Structure2Sql ( $configFile ,&$tableInfo,$sqlOptions=array(),$rootpath = "/ibaconfig/structure")
    272. 

  272.     {
    273. 

  273.         //if it's a string, then it musst be a file, otherwise it's already a config class
    274. 

  274.         $configClass = common::getConfigClass($configFile);
    275. 

  275.  
    276. 

  276.         $dbMasterValues = $configClass->getValues( $rootpath);
    277. 

  277.         // if dont is set, then stop the query building... and return nothing, not used at the moment
    278. 

  278.         if (isset($sqlOptions['dont']) &&  $sqlOptions['dont'])
    279. 

  279.         {
    280. 

  280.             return Null;
    281. 

  281.         }
    282. 

  282.         $queryfields = $dbMasterValues['children'][0].".*";
    283. 

  283.  
    284. 

  284.         $query = " from ".$dbMasterValues['children'][0];
    285. 

  285.         $name = $dbMasterValues['children'][0];
    286. 

  286.  
    287. 

  287.         $dbStructure = $configClass->getValues("$rootpath/$name");
    288. 

  288.  
    289. 

  289.         $queryfields =  $this->getQueryFields($name,$dbStructure,"root",$tableInfo);
    290. 

  290.         $queryfields =  substr($queryfields,1);
    291. 

  291.         $andcondition = "";
    292. 

  292.         if (isset($dbMasterValues["recursive"]) && $dbMasterValues["recursive"] == "tree") {
    293. 

  293.  
    294. 

  294.             include_once("bitlib/SQL/Tree.php");
    295. 

  295.             $t = new sql_tree($this->db);
    296. 

  296.             $t->tablename = $name;
    297. 

  297.             $data = explode(" , ",$dbStructure['fields']);
    298. 

  298.             $sitemapStartTreeID = $this->getAttrib("treeStartID");
    299. 

  299.             if (isset($sqlOptions['start_id'])) {
    300. 

  300.             
    301. 

  301.             } elseif (  $sitemapStartTreeID  ) {
    302. 

  302.                     $sqlOptions['start_id'] = $sitemapStartTreeID ;
    303. 

  303.             } elseif  ( isset($dbMasterValues["start_id"]) ) {

  304.                     $sqlOptions['start_id'] = $dbMasterValues["start_id"];
    305. 

  305.             } else {
    306. 

  306.                 $sqlOptions['start_id'] = 1;
    307. 

  307.             }
    308. 

  308.             $query = $t->children_query_byname(array("ID"=>$sqlOptions['start_id']),$data,True);
    309. 

  309.         }
    310. 

  310.         
    311. 

  311.         elseif (isset($dbMasterValues["recursive"]) && $dbMasterValues["recursive"] == "parents") {
    312. 

  312.  
    313. 

  313.             include_once("bitlib/SQL/Tree.php");
    314. 

  314.             $t = new sql_tree($this->db);
    315. 

  315.            $t->tablename = $name;
    316. 

  316.             $data = explode(" , ",$dbStructure['fields']);
    317. 

  317.             $query = $t->supers_query_byname(array("ID"=>$sqlOptions[start_id]),$data,False);
    318. 

  318.  
    319. 

  319.         }        
    320. 

  320.         
    321. 

  321.         elseif (isset($dbMasterValues["recursive"]) && $dbMasterValues["recursive"] == "children") {
    322. 

  322.  
    323. 

  323.             include_once("bitlib/SQL/Tree.php");
    324. 

  324.             $t = new sql_tree($this->db);
    325. 

  325.             $t->tablename = $name;
    326. 

  326.             $data = explode(" , ",$dbStructure['fields']);
    327. 

  327.             $query = $t->children_query_byname(array("ID"=>$sqlOptions[start_id]),$data);
    328. 

  328.         }
    329. 

  329.         else {
    330. 

  330.             $path = $rootpath;
    331. 

  331.             
    332. 

  332.             while (isset($dbStructure["children"]) &&  is_array($dbStructure["children"]))
    333. 

  333.  
    334. 

  334.             {
    335. 

  335.                 
    336. 

  336.                 $path = $path."/".$name;
    337. 

  337.  
    338. 

  338.                 $parentname = $name;
    339. 

  339.  
    340. 

  340.                 if (! isset($dbStructure["nofields"]) )
    341. 

  341.                {
    342. 

  342.                     $xmlparent = $parentname;
    343. 

  343.                 }
    344. 

  344.  
    345. 

  345.                 $parentdbStructure = $dbStructure;
    346. 

  346.                 foreach($dbStructure["children"] as $child) {
    347. 

  347.                     $name = $child;
    348. 

  348.  
    349. 

  349.  
    350. 

  350.                     $dbStructure = $configClass->getValues( "$path/$name");
    351. 

  351.  
    352. 

  352.                     $queryfields = $queryfields. $this->getQueryFields($child,$dbStructure,$xmlparent,$tableInfo);
    353. 

  353.  
    354. 

  354.  
    355. 

  355.                     if (! isset($dbStructure["thatfield"])) { $dbStructure["thatfield"] = "ID";}
    356. 

  356.                     if (! isset($dbStructure["thisfield"])) { $dbStructure["thisfield"] = "ID";}
    357. 

  357.                     $query = $query ." left join ".$name. " on ($name.".$dbStructure["thisfield"]." = $parentname.".$dbStructure["thatfield"];
    358. 

  358.                     if (isset($dbStructure["objectfield"]) )
    359. 

  359.                     {
    360. 

  360.                         $query = "$query and $parentname.".$dbStructure["objectfield"]." = '$name'";
    361. 

  361.                     }
    362. 

  362.                     elseif (isset($parentdbStructure["objectfield2"]))
    363. 

  363.                     {
    364. 

  364.                         $query = "$query and $parentname.".$parentdbStructure["objectfield2"]." = '$name'";
    365. 

  365.  
    366. 

  366.                     }
    367. 

  367.  
    368. 

  368.                     $query .= ")";
    369. 

  369.                     
    370. 

  370.                     if (isset($dbStructure["activefield"]))
    371. 

  371.                     {
    372. 

  372.                     $activequery = "$name.".$dbStructure["activefield"] ."= 1 or isnull($name.".$dbStructure["activefield"] .")";
    373. 

  373.                     if (isset($dbStructure["activefromfield"]))
    374. 

  374.                     {
    375. 

  375.                         $activequery .= " or (( $child.".$dbStructure["activefield"] ." = 2) and (($child.".$dbStructure["activefromfield"] ."< now()  or $child.".$dbStructure["activetillfield"] ." = 0) and ($child.".$dbStructure["activetillfield"] ."> now() or $child.".$dbStructure["activetillfield"] ." = 0)))";
    376. 

  376.                     }
    377. 

  377.                                 
    378. 

  378.                                 
    379. 

  379.                      $andconditions[] = $activequery;                        
    380. 

  380.                 }
    381. 

  381.                     
    382. 

  382.                     
    383. 

  383.                 }
    384. 

  384.                 // if stopattable is set, then stop the query building...
    385. 

  385.                 if (isset($sqlOptions["stopattable"]) && $sqlOptions["stopattable"] == $name) {
    386. 

  386.                     break;
    387. 

  387.                 }
    388. 

  388.                 // debug::print_rp($dbStructure);
    389. 

  389.             }
    390. 

  390.             $query = "select $queryfields".$query;
    391. 

  391.             if (!isset($sqlOptions["ShowAll"]) && isset($andconditions) && is_array($andconditions))
    392. 

  392.             {
    393.
