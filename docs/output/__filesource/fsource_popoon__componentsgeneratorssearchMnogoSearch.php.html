  127.     function _parse_dsn() {
    128. 

  128.         if(preg_match("/(.*\/)(\?dbmode=(.*))/i",$this->dsn,$match)) {
    129. 

  129.             if($match[1]) {
    130. 

  130.                 $this->set_DBAddr($match[1]);
    131. 

  131.             }
    132. 

  132.             
    133. 

  133.             if(ereg("(single|multi|crc|crc-multi)",$match[3])) {
    134. 

  134.                 $this->set_DBMode($match[3]);
    135. 

  135.             }
    136. 

  136.         } 
    137. 

  137.                 
    138. 

  138.         return TRUE;

  167.     function _result() {
    168. 

  168.         $this->Result = array();
    169. 

  169.         if (is_resource($this->_result) && $this->ResFound > 0) {
    170. 

  170.             
    171. 

  171.             for($i=0; $i<$this->ResRows; $i++) {
    172. 

  172.                 $row = array();
    173. 

  173.                 foreach($this->ResField as $field) {
    174. 

  174.                     if($fval = udm_get_res_field($this->_result,$i,$field[1])) {
    175. 

  175.                         $row[$field[0]] = $fval;
    176. 

  176.                     }
    177. 

  177.                 }
    178. 

  178.                 
    179. 

  179.                 $row['nr'] = ($i+1);
    180. 

  180.                 $row['a_view']=preg_replace("/\?*".session_name()."=[a-zA-Z0-9](.*)/","",$row['url']);
    181. 

  181.                 if (isset($row['mod'])) {
    182. 

  182.                     $row['date'] = date("d.m.y",$row['mod']);
    183. 

  183.                 }
    184. 

  184.                 
    185. 

  185.  
    186. 

  186.                 array_push($this->Result,$row);    
    187. 

  187.             }    
    188. 

  188.         }
    189. 

  189.         
    190. 

  190.         return $this->Result;
    191.

  202.     function _query($query) {

  203.         $this->query = $this->_eval_query($query);
    204. 

  204.         if (!empty($this->query) && is_resource($this->MRes)) {
    205. 

  205.             if($this->_result = udm_find($this->MRes,$this->query)) {

    212. 

  212.                 
    213. 

  213.                 return TRUE;
    214. 

  214.             }
    215. 

  215.             return array("error"=>"mnogosearch:" . udm_Error($this->MRes));;
    216. 

  216.         } else {
    217. 

  217.  
    218. 

  218.             return FALSE;
    219. 

  219.         }

  230.     function msAllocate($dsn=NULL) {
    231. 

  231.         if ($dsn == NULL) {
    232. 

  232.             return array("error" => 'No DSN for mnogosearch provided.');
    233. 

  233.         } else if (!extension_loaded('mnogosearch')) {
    234. 

  234.             return array("error" => 'mnogosearch extension is not loaded/installed.');
    235. 

  235.         } else {
    236. 

  236.             $this->set_dsn($dsn);
    237. 

  237.             $this->_parse_dsn($this->dsn);
    238. 

  238.         }
    239. 

  239.             
    240. 

  240.         
    241. 

  241.         // MnogoSearch <= 3.1.2 doesn't
    242. 

  242.         // accepts dbmode in dsn
    243. 

  243.         if ($this->apiversion <= 30130) {
    244. 

  244.             if(!$this->MRes = udm_alloc_agent($this->DBAddr,$this->DBMode)) {
    245. 

  245.                 return array("error" => 'mnogosearch could not allocate Agent');
    246. 

  246.             }
    247. 

  247.         } else {
    248. 

  248.             if(!$this->MRes = udm_alloc_agent($this->DBAddr."?dbmode=".$this->DBMode)) {
    249. 

  249.                 return array("error" => 'mnogosearch could not allocate Agent');
    250. 

  250.             }
    251. 

  251.         }
    252. 

  252.         return $this->MRes;
    253.
