  153.         $this->sitemap = &$sitemap;

  154.         //determine cache container

  155.         if(isset($pipelineAttribs['container'])) {
    156. 

  156.             $containerType = $pipelineAttribs['container'];

  157.         }
    158. 

  158.         else {
    159. 

  159.             $containerType = $this->defaultContainer;

  160.         }
    161. 

  161.  
    162. 

  162.         if(!isset($this->cacheHandlers[$containerType]) || !is_object($this->cacheHandlers[$containerType])){
    163. 

  163.             $this->cacheHandlers[$containerType] =& new Cache($containerType, $this->getContainerOptions($containerType, $pipelineAttribs));

  178.     function init(&$component){
    179. 

  179.         
    180. 

  180.         //only work with components implementing the 'cachable' interface
    181. 

  181.         if(!$this->implementscachable($component)) {
    182. 

  182.             $this->iscachable = false;

  183.             popoon::raiseError("You're trying to use a non-cachable component inside of a pipeline marked cachable.\n".
    184. 

  184.                                "This should be supported somehow but isn't implemented yet. Sorry.",
    185. 

  185.                                POPOON_ERROR_WARNING,
    186. 

  186.                                __FILE__, __LINE__, null);
    187. 

  187.             return(false);
    188. 

  188.         }
    189. 

  189.         else{
    190. 

  190.             $this->currentComponent =& $component;
    191. 

  191.             $this->lastCacheKey = $this->currentCacheKey;
    192. 

  192.             $this->componentTypes[] = $this->currentComponent->attribs['type'];
    193. 

  193.             $this->currentCacheKey = $component->generateKey($component->attribs, $this->lastCacheKey);
    194. 

  194.             $this->setCacheGroup();
    195. 

  195.             return(true);

  201.     /**
    202. 

  202.      * Try to get cached component output
    203. 

  203.      *
    204. 

  204.      * Tries to get a validityObject for the current request. If there's one and if the component does verify it's validity,
    205. 

  205.      * the cached content is fetched and returned. Works only with Components implementing the 'cachable' interface, of course.
    206. 

  206.      * In any other case than success returns null.
    207. 

  207.      *
    208. 

  208.      * @return   string   cached content
    209. 

  209.      * @access   private
    210. 

  210.      * @see implementscachable(), loadValidityObject(), checkValidity(), loadCachedContent()
    211. 

  211.      */
    212. 

  212.     function _load($what = 'content'){
    213. 

  213.  
    214. 

  214.         $component =& $this->currentComponent;
    215. 

  215.                    
    216. 

  216.         if(!$this->miss && !is_null($validityObject = $this->loadValidityObject())){  //there's a candidate in the cache!
    217. 

  217.             //test if this candidate is a valid one    
    218. 

  218.             if($component->checkValidity($validityObject)){
    219. 

  219.                 //if so, we do really have a valid cache hit!
    220. 

  220.                 if($what == 'content') {
    221. 

  221.                      return($this->loadCachedContent());
    222. 

  222.                 }
    223. 

  223.                 else {
    224. 

  224.                     return($this->currentCacheKey);
    225. 

  225.                 }
    226. 

  226.             }
    227. 

  227.         }
    228. 

  228.         //in any other case
    229. 

  229.         $this->miss = true;
    230. 

  230.         return(null);
    231. 

  231.     }

  233.     /**
    234. 

  234.      * Load content associated with last cache key
    235. 

  235.      *
    236. 

  236.      * When we come to a point in the pipeline where we need the output of the last component, we call loadLast().
    237. 

  237.      * It resets the cache object to the state it was when handling the last component, fetches the content wanted and
    238. 

  238.      * sets back the state of this object.
    239. 

  239.      *
    240. 

  240.      * @param  $target  reference to the variable you wan't to have the content you're looking for put into.
    241. 

  241.      */
    242. 

  242.     function loadLast(&$target){         
    243. 

  243.  
    244. 

  244.         //get last componentType
    245. 

  245.         $pos = count($this->componentTypes) - 2; if($pos < 0) $pos = 0; $lastComponentType = $this->componentTypes[$pos];
    246. 

  246.  
    247. 

  247.         //backup currentCacheKey
    248. 

  248.         $t = $this->currentCacheKey;
    249. 

  249.  
    250. 

  250.         //last is now current
    251. 

  251.         $this->currentCacheKey = $this->lastCacheKey;
    252. 

  252.  
    253. 

  253.         //also set back currentCacheGroup
    254. 

  254.         $this->setCacheGroup($lastComponentType, $this->lastCacheKey);
    255. 

  255.  
    256. 

  256.         /*load it
    257. 

  257.            but just, if target is_null, if it's not null, we already have something in 
    258. 

  258.            $this->xml. This is mostly set, when a component before also had to regenerate
    259. 

  259.            its data and can be mostly seen when we create a whole pipeline (eg. the generator
    260. 

  260.             loads xml-data, then passes that to a transformer. There is no need to read the
    261. 

  261.             content from the cache, if something like that happens. Moreover it leads to
    262. 

  262.             problems with libxslt2/libxml internal handling, as it looses some info when 
    263. 

  263.             we serialize and deserialize xml-data (this has to be tested more thoroughly)
    264. 

  264.             *by chregu*
    265. 

  265.         */
    266. 

  266.         
    267. 

  267.         if (is_null($target)) {
    268. 

  268.             $target = $this->loadCachedContent();
    269. 

  269.         }
    270. 

  270.        
    271. 

  271.         //reset all
    272. 

  272.         $this->currentCacheKey = $t;
    273. 

  273.         $this->setCacheGroup();
    274. 

  274.     }

  309.      * Store component generated data

  310.      *

  311.      * Tells the current component to produce the data and call the methods to store that content and the according validityObject.
    312. 

  312.      *
    313. 

  313.      * @param   string   $launchMethod     Method to call to make the component generate the content
    314. 

  314.      * @param   bool     $printsDirectly  Indicates wether $launchMethod prints the content directly (as e.g. the readers do) or wether we have to get the content from $content
    315. 

  315.      * @access  public
    316. 

  316.      * @see storeContent(), $storeValidityObhect()
    317. 

  317.      */
    318. 

  318.     function store($content = null, $launchMethod = null, $param = null, $printsDirectly = false){
    319. 

  319.         
    320. 

  320.         if(!$this->iscachable) return(false); //we could also support a mix of cachable and non-cachable compos.,
    321. 

  321.         
    322. 

  322.         if($printsDirectly) {
    323. 

  323.             ob_start();
    324. 

  324.         } 
    325. 

  325.  
    326. 

  326.         if(!is_null($launchMethod) && method_exists($this->currentComponent, $launchMethod)) {
    327. 

  327.               $retVal = $this->currentComponent->$launchMethod($param);
    328. 

  328.         }
    329. 

  329.         else { /*add error handling here?*/ }
    330. 

  330.         
    331. 

  331.         if($printsDirectly){  // if $printsDirectly, then it must be some kind of serializer, so we need the headers on output
    332. 

  332.             $this->sitemap->printHeader();
    333. 

  333.             $content = ob_get_contents();
    334. 

  334.             ob_end_flush();
    335. 

  335.         }
    336. 

  336.         // else{ sitemap::var2XMLString($content); }
    337. 

  337.         if ( strtolower(get_class($content)) == "domdocument")
    338. 

  338.         {
    339. 

  339.             $content = $content->dumpmem();
    340. 

  340.         }
    341. 

  341.         
    342. 

  342.         $done = $this->storeContent($content) &&  $this->storeValidityObject()  ;   //only store validityObject after content was saved successfully.
    343. 

  343.         if(!$done)  popoon::raiseError('Could not write to cache! Is cache_dir writable? Is your component outputting?',
    344. 

  344.                                        POPOON_ERROR_WARNING,
    345. 

  345.                                        __FILE__, __LINE__, null);
    346. 

  346.         return($content);
    347. 

  347.         
    348. 

  348.     }
