  76.             // mmmh, if params are sent seperated with  &amp;, php does some funny stuff with _REQUEST

    78. 

  78.             if (strpos($key,"amp;") === 0) {
    79. 

  79.                 unset($querystring[$key]);

  80.                 $key = str_replace("amp;","",$key);
    81. 

  81.             }
    82. 

  82.             $querystring["$key"] = "<![CDATA[".$value."]]>";
    83. 

  83.         }
    84. 

  84.         /* set options for structure2xml */

  86.                        'querystring' => $querystring,

    88. 

  88.                    );

    90.

    93. 

  93.             foreach($searches as $section => $query)
