  417.     function getForumId(&$URL){
    418. 

  418.  
    419. 

  419.  
    420. 

  420.         // "/tutorials/slideml" should be the same as "/tutorials/slideml/" and /tutorial//slideml
    421. 

  421.         if (substr($URL, -1) == '/'){
    422. 

  422.  
    423. 

  423.             $URL = substr($URL, 0, -1);
    424. 

  424.  
    425. 

  425.         }
    426. 

  426.  
    427. 

  427.         $URL = preg_replace('#//+#', '/', $URL);
    428. 

  428.  
    429. 

  429.         //figure out table names
    430. 

  430.         $forumTable = $this->dbInfo['table_prefix'] . '_forums';
    431. 

  431.         $messageTable = $this->dbInfo['table_prefix'] . '_messages';
    432. 

  432.  
    433. 

  433.         //try to fetch forumId. We use the column "name" to store the URL in.
    434. 

  434.         if($this->checkPhorumDbError($this->forumId  =
    435. 

  435.                                      $this->db->getOne("SELECT forum_id FROM $forumTable WHERE name = '$URL'") , __LINE__)) {
    436. 

  436.  
    437. 

  437.             //if there's no entry, we'll create one.
    438. 

  438.             if(is_null($this->forumId)) {
    439. 

  439.  
    440. 

  440.                 //set list_length for insertion
    441. 

  441.                 if(!$list_length = $this->getParameterDefault('listlength')){
    442. 

  442.  
    443. 

  443.                     $list_length = $this->defaultListLength;
    444. 

  444.  
    445. 

  445.                 }
    446. 

  446.  
    447. 

  447.                 //if there's no forum_id for this URL, we'll enter a new one
    448. 

  448.                 $this->forumId = $this->db->nextID($forumTable);
    449. 

  449.  
    450. 

  450.                 $this->checkPhorumDbError($this->db->query("INSERT INTO $forumTable SET".
    451. 

  451.                                                            '  forum_id       = ' . $this->forumId .
    452. 

  452.                                                            ', name           = "'. $this->URL .'"'.
    453. 

  453.                                                            ', list_length    ='  . $list_length   .
    454. 

  454.                                                            ', threaded       ='  . (int) $this->threaded .
    455.
