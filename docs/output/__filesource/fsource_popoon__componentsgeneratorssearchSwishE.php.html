  164. 	* Evaluate Query

  165. 	* Escape malicious elements and catch if there

  166. 	* are already boolean elements (and|or|not|near) or

  172.     function _eval_query($query) {
    173. 

  173.         
    174. 

  174.         $query = escapeshellcmd($query);
    175. 

  175.         
    176. 

  176.         if (preg_match("/\(|\)|\| and | or | not | near /",$query,$matches)) {
    177. 

  177.             $this->setter('_mode','boolean','string');
    178. 

  178.         } else {
    179. 

  179.             if (!empty($this->_mode)) {
    180. 

  180.                     
    181. 

  181.                 $query = trim($query);
    182. 

  182.                 
    183. 

  183.                 switch ($this->_mode) {
    184. 

  184.                     case "all":
    185. 

  185.                         $query = preg_replace("/[[:space:]]/","/ AND /",$query);
    186. 

  186.                     break;
    187. 

  187.                     case "any":
    188. 

  188.                         $query = preg_replace("/[[:space:]]/","/ OR /",$query);
    189. 

  189.                     break;
    190. 

  190.                     case "phrase":
    191. 

  191.                         $query = "\"".$query."\"";
    192. 

  192.                     break;
    193. 

  193.                 }
    194. 

  194.             }        
    195. 

  195.         }
    196. 

  196.

  236.         if(is_string($exec)) {

  237.             if(($p = popen($exec,"r"))!=FALSE) {

    246.

  261.     function _parse_result($result) {
    262. 

  262.     
    263. 

  263.         preg_match("/\# Number of hits: ([0-9]{1,})/i",$result,$match);
    264. 

  264.         $this->ResFound = intval($match[1]);
    265. 

  265.         preg_match("/\# Run time: ([0-9]{1,}.[0-9]{1,})/i",$result,$match);
    266. 

  266.         $this->ResTime = $match[1];
    267. 

  267.         
    268. 

  268.         $result = preg_replace("/\# (.*)\\n/","",$result);
    269. 

  269.         $result = explode("\n",$result);
    270. 

  270.         $this->Result = array();
    271. 

  271.         
    272. 

  272.         foreach ($result as $res) {
    273. 

  273.             if(strlen($res)>1) {
    274. 

  274.                 $res = explode($this->_delim,$res);
    275. 

  275.                 $row = array();
    276. 

  276.                 foreach($this->_resp as $k=>$param) {
    277. 

  277.                     $row[$param] = $res[$k];
    278. 

  278.                 }
    279. 

  279.                 
    280. 

  280.                 array_push($this->Result,$row);
    281. 

  281.             }
    282. 

  282.         }
    283. 

  283.         
    284. 

  284.         return $this->Result;
    285.
