  115.     function getDbFromDsn  ($dsn) {
    116. 

  116.     
    117. 

  117.         if (is_string($dsn))
    118. 

  118.         {
    119. 

  119.             //check if we have a handle already in the global pool.
    120. 

  120.             // if yes, return that, thereforee we don't have to create
    121. 

  121.             // a new one, which is quite expensive 
    122. 

  122.             if (isset($GLOBALS["_POPOON_globalContainer"]->DB_Handles[$dsn]))
    123. 

  123.             { 
    124. 

  124.                 return $GLOBALS["_POPOON_globalContainer"]->DB_Handles[$dsn];
    125. 

  125.             }
    126. 

  126.             
    127. 

  127.             include_once ("DB.php");
    128. 

  128.  
    129. 

  129.             $GLOBALS["_POPOON_globalContainer"]->DB_Handles[$dsn] = DB::Connect($dsn);
    130. 

  130.             $db = &$GLOBALS["_POPOON_globalContainer"]->DB_Handles[$dsn];
    131. 

  131.             if (DB::isError($db))
    132. 

  132.             {
    133. 

  133.  
    134. 

  134.                 return common::raiseError("could not connect to database: <br />\n".$GLOBALS["_POPOON_globalContainer"]->DB_Handles[$dsn]->getMessage(),__FILE__,__LINE__,null);                
    135. 

  135.             }
    136. 

  136.             
    137. 

  137.         }
    138. 

  138.  
    139. 

  139.         elseif (is_object($dsn) && DB::isError($dsn))
    140. 

  140.         {
    141. 

  141.                return popoon::raiseError("could not connect to database:<br />\n".$GLOBALS["_POPOON_globalContainer"]->DB_Handles[$dsn]->getMessage(),POPOON_ERROR_FATAL);
    142. 

  142.         }
    143. 

  143.  
    144. 

  144.         // if parent class is db_common, then it's already a connected identifier
    145. 

  145.         elseif (get_parent_class($dsn) == "db_common")
    146. 

  146.         {
    147. 

  147.             $db = $dsn;
    148. 

  148.         }
    149. 

  149.         return $db;

  162.     function get_xml($xml,$instring = "    ") {
    163. 

  163.             $xml = preg_replace("/(\>)\n/","$1",$xml);
    164. 

  164.             $xml = preg_replace("/\>\s*\</",">\n<",$xml);
    165. 

  165.  
    166. 

  166.             $axml = explode("\n",$xml);
    167. 

  167.  
    168. 

  168.             $indent=-1;
    169. 

  169.             $xmls = "";
    170. 

  170.             foreach ($axml as $key => $value) {
    171. 

  171.  
    172. 

  172.                 if (preg_match("/<[^\/{1}]/",$value)) {
    173. 

  173.                     $indent++;
    174. 

  174.                 }
    175. 

  175.             if ($indent < 0)
    176. 

  176.                 $indent = 0;
    177. 

  177.                 $xmls .= str_repeat($instring,$indent);
    178. 

  178.                 if (preg_match("/\<\//",$value) || preg_match("/\/\>/",$value)|| preg_match("/-->/",$value)) {
    179. 

  179.                     $indent--;
    180. 

  180.                 }
    181. 

  181.                 $xmls .= trim($value)."\n";
    182. 

  182.             }
    183. 

  183.            return $xmls;
    184. 

  184.     }
