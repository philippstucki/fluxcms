  36.     var $treeFields = array("ID","fulluri","onedocumentonly","structure","xslt","foreignsectionid","button","includechildren","includesibling","includeparent","l","r");
    37. 

  37.     var $defaultStructure = "simple.xml";
    38. 

  38.     var $defaultXslt = "root.xsl";
    39. 

  39.     var $db = false;
    40. 

  40.     var $structureDir = null;
    41. 

  41.     var $defaultLang = "en";
    42. 

  42.     var $xslDir = null;
    43. 

  43.     var $choosenLang = "";
    44. 

  44.     var $defaultPageExtension = "html";
    45. 

  45.     var $checkModes = "";
    46. 

  46.     /**
    47.

    48. 

  48.     *
    49. 

  49. 	*/

  50.     function action_bitfluxcms(&$sitemap) {
    51.

    52. 

  52.     }
    53. 

  53.  
    54. 

  54.     function init($attribs) {
    55. 

  55.           parent::init($attribs);
    56. 

  56.     }

  58.     function act() {
    59. 

  59.         include_once("bitlib/SQL/Tree.php");        
    60. 

  60.         if ($this->structureDir == null) {
    61. 

  61.             $this->structureDir = BX_PROJECT_DIR."structure/";
    62. 

  62.         }
    63. 

  63.  
    64. 

  64.         if ($this->xslDir == null) {
    65. 

  65.             $this->xslDir = BX_PROJECT_DIR."xsl/";
    66. 

  66.         }
    67. 

  67.         
    68. 

  68.         //some init values to catch E_ALL Warnings
    69. 

  69.         $pageExtension = $this->defaultPageExtension;
    70. 

  70.         $lang = $this->defaultLang;
    71. 

  71.         $page = "";
    72. 

  72.         $DocumentID = "";
    73. 

  73.         $ObjectID = "";
    74. 

  74.         $DocuUri = "";
    75. 

  75.  
    76. 

  76.         $Tree = new SQL_Tree($this->db);
    77. 

  77.         /* this if section is, when you want to get pages with GET Paremeters, for example :
    78. 

  78.             ?index.php?Section=134&Document=23&Object=24&lang=en
    79. 

  79.          */
    80. 

  80.         if (isset($_GET["Section"])) {
    81. 

  81.             $DocuIDonly =$_GET["Document"];
    82. 

  82.             $DocumentID="Document".$_GET["Document"];
    83. 

  83.             if (isset ($_GET["Object"])) {
    84. 

  84.                 $ObjectID="".$_GET["Object"];
    85. 

  85.             }
    86. 

  86.             $r = array("ID"=>$_GET["Section"]);
    87. 

  87.             if (isset($_GET["lang"]))  {
    88. 

  88.                 $lang = $_GET["lang"];
    89. 

  89.             }
    90. 

  90.  
    91. 

  91.         }
    92. 

  92.         else {
    93. 

  93.             //here comes the whole path to section/document conversion stuff
    94. 

  94.             $path = $this->getAttrib("path");
    95. 

  95.             if (is_null($path)) {
    96. 

  96.                 if (!isset($_GET["path"])) {
    97. 

  97.                     $path = "";
    98. 

  98.                 } else {
    99. 

  99.                     $path = $_GET["path"];
    100. 

  100.                 }
    101. 

  101.             }
    102. 

  102.  
    103. 

  103.             $path= preg_replace("#/+#","/",$path);
    104. 

  104.             $path= preg_replace("#^/#","",$path);
    105. 

  105.             $path= preg_replace("#/$#","",$path);
    106. 

  106.  
    107. 

  107.             $pages = explode("/",$path);
    108. 

  108.  
    109. 

  109.             /* get language
    110. 

  110.              * the language is always taken from the first "dir" in the uri, if it has exactly 2 letters
    111. 

  111.              */
    112. 

  112.             if (preg_match("/^([a-z]{2})$/",$pages[0],$match)) {
    113. 

  113.                 $lang = $match[1];
    114. 

  114.                 array_shift($pages);
    115. 

  115.             }
    116. 

  116.             //if choosenLang is defined, set this as language. always.
    117. 

  117.             if ($this->choosenLang) {
    118. 

  118.                 $lang = $this->choosenLang;
    119. 

  119.             }
    120. 

  120.             $mode = "";
    121. 

  121.             //check Modes and strip them from the beginnin
    122. 

  122.             if ($this->checkModes) {
    123. 

  123.                 $modes = explode(" ",$this->checkModes);
    124. 

  124.                 if (in_array($pages[0],$modes)) {
    125. 

  125.                     $mode = array_shift($pages);
    126. 

  126.                 }
    127. 

  127.             }
    128. 

  128.             
    129. 

  129.             /* get the fulluri at this point. it's with the page stuff */
    130. 

  130.             $fulluri = implode("/",$pages);
    131. 

  131.             
    132. 

  132.             /* get page, pageExtension, DocuUri, $ObjectID
    133. 

  133.              "$fulluri/$DocuUri_$ObjectID.$pageExtension" 
    134. 

  134.              $page = $DocuURI_$ObjectID.$pageExtension
    135. 

  135.              */
    136. 

  136.             if (isset($pages[count($pages)-1]) && preg_match("/(.*)(_([0-9]*))*\.([a-z0-9A-Z]{3,})$/U",$pages[count($pages)-1],$match)) {
    137. 

  137.                 $page =$pages[count($pages)-1];
    138. 

  138.                 $DocuUri = $match[1];
    139. 

  139.                 $pageExtension = $match[4];
    140. 

  140.                 if (isset($match[3])) {
    141. 

  141.                     $ObjectID = $match[3];
    142. 

  142.                 }
    143. 

  143.                 /* take this last part away, it's not needed to find the correct section */
    144. 

  144.                 array_pop($pages );
    145. 

  145.             }
    146. 

  146.             // get the path for the requested section
    147. 

  147.             $fulluriSection = implode("/",$pages);
    148. 

  148.             $r = array("fulluri"=>$fulluriSection);
    149. 

  149.         }
    150. 

  150.         
    151. 

  151.         //get the nested-set-tree with the found $uri
    152. 

  152.         $result = $Tree->supers_byname($r,$this->treeFields);
    153. 

  153.  
    154. 

  154.         /* if nothing was found, we hit an uri, which has no corresponding
    155. 

  155.            section, so return null, we can handle this then in the sitemap
    156. 

  156.            */
    157. 

  157.         if (! $result) {
    158. 

  158.             return null;
    159. 

  159.         }
    160. 

  160.  
    161. 

  161.         // take the last result
    162. 

  162.         $Srow = $result[count($result)];
    163. 

  163.         
    164. 

  164.         /* onceDocumentONly is not really used anymore
    165. 

  165.            it was intended to output only one document in the output instead
    166. 

  166.            of all documents, which belong to the requested section.
    167. 

  167.            With caching this is rather obsolet (except you have really big documents)
    168. 

  168.         */
    169. 

  169.         
    170. 

  170.         $oneDocumentOnly = $Srow["onedocumentonly"];
    171. 

  171.  
    172. 

  172.         // set default files
    173. 

  173.         $Srow["structure"] = $this->defaultStructure;
    174. 

  174.         $Srow["xslt"] = $this->defaultXslt;
    175. 

  175.  
    176. 

  176.         /* loop through all found section from top to bottom
    177. 

  177.            and set the respective values, which can be inherited
    178. 

  178.         */
    179. 

  179.         foreach ($result as $row) {
    180. 

  180.             if (!(empty($row["xslt"]))) {
    181. 

  181.                 $Srow["xslt"] = $row["xslt"];
    182. 

  182.             }
    183. 

  183.             if (!(empty($row["structure"]))) {
    184. 

  184.                 $Srow["structure"] = $row["structure"];
    185. 

  185.             }
    186. 

  186.             if (!(empty($row["button"]))) {
    187. 

  187.                 $Srow["HighlightID"] = $row["ID"];
    188. 

  188.             }
    189. 

  189.         }
    190. 

  190.         
    191. 

  191.         $pageWhere = "Section.ID = $Srow[ID]  ";
    192. 

  192.  
    193. 

  193.         $section_left = $Srow['l'];
    194. 

  194.         $section_right = $Srow['r'];
    195. 

  195.  
    196. 

  196.         // if children sections should be included as well
    197. 

  197.         if ($Srow["includechildren"]) {
    198. 

  198.             $pageWhere .= " or Section.foreignsectionid = $Srow[ID]";
    199. 

  199.         }
    200. 

  200.         // if sibling sections should be included as well
    201. 

  201.         if ($Srow["includesibling"]) {
    202. 

  202.             $pageWhere .= " or Section.foreignsectionid = $Srow[foreignsectionid]";
    203. 

  203.         }
    204. 

  204.         // if the parent section should be included as well
    205. 

  205.         if ($Srow["includeparent"]) {
    206. 

  206.             $pageWhere .= " or Section.ID = $Srow[foreignsectionid] ";
    207. 

  207.         }
    208. 

  208.  
    209. 

  209.         if (strlen($fulluriSection) > 0) {
    210. 

  210.             $fulluriSection .= "/";
    211. 

  211.         }
    212. 

  212.             return array(      
    213. 

  213.                             "xsltFile" => $this->xslDir.$Srow["xslt"],
    214. 

  214.                             "structureFile" => $this->structureDir.$Srow["structure"],
    215. 

  215.                             "lang" => $lang,
    216. 

  216.                             "page" => $page,
    217. 

  217.                             "pageExtension" => $pageExtension,
    218. 

  218.                             "pageWhere" => $pageWhere,
    219. 

  219.                                          
    220. 

  220.                             "fulluri" => $fulluri,
    221. 

  221.                             "fulluriSection" => $fulluriSection,
    222. 

  222.                             "SectionID"=>"Section".$Srow["ID"],
    223. 

  223.                             "DocumentID"=>$DocumentID,
    224. 

  224.                             "DocumentUri" => $DocuUri,
    225. 

  225.                             "ObjectID"=>$ObjectID,
    226. 

  226.                             "Section_left" => $section_left,
    227. 

  227.                             "Section_right" => $section_right,
    228. 

  228.                             "mode" => $mode,
    229. 

  229.                               );
    230. 

  230.      
    231. 

  231.     }
    232. 

  232. }
