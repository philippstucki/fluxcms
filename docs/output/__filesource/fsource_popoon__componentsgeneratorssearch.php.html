  63.        /**
    64. 

  64.     * Initial Call / Interface to sitemap.php
    65. 

  65. 	*
    66. 

  66. 	* loads Search Module and gets Parameter
    67. 

  67. 	* from Sitemap. Sets Classvars
    68. 

  68. 	*
    69. 

  69. 	* @param array $attr 
    70. 

  70. 	* @return bool true|false

    72. 

  72.     function init($attr) {
    73. 

  73.         $this->attribs = $attr;
    74. 

  74.         
    75. 

  75.         /* Load appropriate Searchmodule , default MnogoSearch */
    76. 

  76.         $mod = $this->getParameter('default','searchModule');
    77. 

  77.         $mod = (empty($mod))?'MnogoSearch':$mod;
    78. 

  78.         $ret = $this->loadSearchModule($mod);
    79. 

  79.         if(is_array($ret) && isset($ret['error'])) {
    80. 

  80.             Popoon::raiseError($ret['error']);
    81. 

  81.         }
    82. 

  82.         
    83. 

  83.         $this->query = $this->getParameter('default','query');
    84. 

  84.         $this->uri = $this->getParameter('default','requestUri');
    85. 

  85.         $this->currpage = $this->getParameter('module','CurrPage');
    86. 

  86.         $this->numrows = $this->getParameter('module','NumRows');

  87.         return TRUE;
    88.

  91.     /**
    92. 

  92. 	* Start Query Processing / generate XML
    93. 

  93. 	* Interface to sitemamp.php
    94. 

  94. 	* takes (string) $xml from sitemap.php by reference
    95. 

  95. 	* xml-string from buildXML() gets appended to $xml

  96. 	* @param string xml

    99. 

  99.     function DomStart(&$xml) {
    100. 

  100.         $res = $this->Search->doQuery($this->query);
    101. 

  101.         if (is_array($res) && isset($res['error'])) {
    102. 

  102.             $xml = $this->buildErrorXml($res['error']);
    103. 

  103.         } else {
    104. 

  104.             $this->maxpages = $this->getMaxPages();
    105. 

  105.             $this->result = $this->Search->getResult();
    106. 

  106.             //var_dump($this->result);
    107. 

  107.             if (is_array($this->result) && isset($this->result['error'])) {
    108. 

  108.             
    109. 

  109.                 $xml = $this->buildErrorXml($this->result['error']);
    110. 

  110.             } else {
    111. 

  111.                 
    112. 

  112.                 $xml = $this->buildXml();
    113. 

  113.                 return TRUE;
    114. 

  114.             }
    115. 

  115.         }
    116. 

  116.     }

    120. 

  120. 	* Load appropriate search module
    121. 

  121. 	*
    122. 

  122. 	* Search module has to be a class in generators/search/
    123. 

  123. 	* parameter $mod is name of File/Class to include
    124. 

  124. 	* @param string Modulename
    125. 

  125. 	* @return object Search 
    126. 

  126. 	*/
    127. 

  127.     function loadSearchModule($mod=NULL) {
    128. 

  128.         if($mod==NULL) {
    129. 

  129.             $mod = $this->getParameter('default','searchModule');            
    130. 

  130.         }
    131. 

  131.         
    132. 

  132.         $searchMod = dirname(__FILE__)."/search/".$mod.".php";
    133. 

  133.         if (file_exists($searchMod)) {
    134. 

  134.                 require_once $searchMod;
    135. 

  135.                 $params = $this->getParameter('module');
    136. 

  136.                 eval("\$this->Search =& new ".$mod."(\$params);");
    137. 

  137.         } else {
    138. 

  138.             return array("error"=>"Class $mod not found!");
    139. 

  139.         }
    140. 

  140.         
    141. 

  141.         return $this->Search;
    142. 

  142.     }

  145.     /**
    146. 

  146. 	* returns links for 'previous'/'next' buttons to skim
    147. 

  147. 	* through result pages
    148. 

  148. 	* @param string $what 
    149. 

  149. 	* @return string $request
    150. 

  150. 	*/
    151. 

  151.     function prevNext($what) {
    152. 

  152.         $maxp = (($this->maxpages -1) >=0)?($this->maxpages -1):0;
    153. 

  153.         
    154. 

  154.         if($what == 'next') {
    155. 

  155.             if(($this->currpage +1) <= $maxp) {
    156. 

  156.                 $page = $this->currpage +1;
    157. 

  157.             } else {
    158. 

  158.                 $page = $this->maxpages -1;
    159. 

  159.             }
    160. 

  160.         } elseif($what == 'prev') {
    161. 

  161.             if(($this->currpage -1) >= 1) {
    162. 

  162.                 $page = $this->currpage -1;
    163. 

  163.             } else {
    164. 

  164.                 $page = $this->currpage;
    165. 

  165.             }
    166. 

  166.         }
    167. 

  167.         
    168. 

  168.         $request = $this->getParameter('default','RequestUri');
    169. 

  169.         $request = str_replace("ResCurrPage=","ResCurrPage=".$page,$request);
    170. 

  170.     
    171. 

  171.         return $request;
    172. 

  172.     }

    176. 

  176. 	* Calc. Max Pages displayable according to Results found
    177. 

  177. 	* @return int MaxPages
    178. 

  178. 	*/
    179. 

  179.     function getMaxPages() {
    180. 

  180.         if(!empty($this->Search->ResFound) && $this->numrows > 0) {
    181. 

  181.             return ceil($this->Search->ResFound / $this->numrows);
    182. 

  182.         }
    183. 

  183.     }

  195.     /**
    196. 

  196. 	* Calc. which Document number (count) is the last displayed
    197. 

  197. 	* @return int LastDoc
    198. 

  198. 	*/
    199. 

  199.     function getLastDoc() {
    200. 

  200.         if (($this->currpage +1) < $this->maxpages) {
    201. 

  201.             return ($this->currpage * $this->numrows)+$this->numrows;
    202. 

  202.         } else {
    203. 

  203.             return $this->Search->ResFound;
    204. 

  204.         }
    205. 

  205.     }

  209. 	* Create XML from Search Results

  210. 	* @return string $dom

    212. 

  212.     function BuildXml() {
    213. 

  213.         $dom = domxml_new_doc("1.0");
    214. 

  214.         $root = $dom->add_root('SearchResult');
    215. 

  215.         
    216. 

  216.         if(is_array($this->result)) {
    217. 

  217.             // Total Found Results
    218. 

  218.             $elem = $dom->create_element('Found');
    219. 

  219.             $elem->set_content(utf8_encode($this->Search->ResFound));
    220. 

  220.             $root->append_child($elem);
    221. 

  221.         
    222. 

  222.             // Total displayed on Page
    223. 

  223.             $elem = $dom->create_element('Rows');
    224. 

  224.             $elem->set_content(utf8_encode($this->numrows));
    225. 

  225.             $root->append_child($elem);
    226. 

  226.         
    227. 

  227.             // Current Page
    228. 

  228.             $elem = $dom->create_element('ResCurrPage');
    229. 

  229.             $elem->set_content(utf8_encode($this->currpage));
    230. 

  230.             $root->append_child($elem);
    231. 

  231.             
    232. 

  232.             // NextPage 
    233. 

  233.             $next = $this->prevNext('next');
    234. 

  234.             $elem = $dom->create_element('NextPage');
    235. 

  235.             $elem->set_content($next);
    236. 

  236.             $root->append_child($elem);
    237. 

  237.         
    238. 

  238.             // PreviousPage
    239. 

  239.             $prev = $this->prevNext('prev');
    240. 

  240.             //echo "<br>".$prev;
    241. 

  241.             $elem = $dom->create_element('PrevPage');
    242. 

  242.             $elem->set_content($prev);
    243. 

  243.             $root->append_child($elem);
    244. 

  244.         
    245. 

  245.             // Time used for SearchResult
    246. 

  246.             $elem = $dom->create_element('Time');
    247. 

  247.             $elem->set_content(utf8_encode($this->Search->ResTime));
    248. 

  248.             $root->append_child($elem);
    249. 

  249.         
    250. 

  250.             // Maximum Nr. of Pages
    251. 

  251.             $elem = $dom->create_element('MaxPages');
    252. 

  252.             $elem->set_content(utf8_encode($this->maxpages));
    253. 

  253.             $root->append_child($elem);
    254. 

  254.         
    255. 

  255.             // First Document
    256. 

  256.             $elem = $dom->create_element('FDoc');
    257. 

  257.             $elem->set_content(utf8_encode($this->getFirstDoc()));
    258. 

  258.             $root->append_child($elem);
    259. 

  259.         
    260. 

  260.             // Last Document
    261. 

  261.             $elem = $dom->create_element('LDoc');
    262. 

  262.             $elem->set_content(utf8_encode($this->getLastDoc()));
    263. 

  263.             $root->append_child($elem);
    264. 

  264.         
    265. 

  265.             // Query
    266. 

  266.             $elem = $dom->create_element('Query');
    267. 

  267.             $elem->set_content(utf8_encode($this->query));
    268. 

  268.             $root->append_child($elem);
    269. 

  269.         
    270. 

  270.             // The Results
    271. 

  271.             $elem = $dom->create_element('Results');
    272. 

  272.             $root->append_child($elem);
    273. 

  273.             if(sizeof($this->result)>0 ) {
    274. 

  274.                 foreach($this->result as $key => $val) {
    275. 

  275.                     $res = $dom->create_element('Result');
    276. 

  276.                         foreach($val as $k => $v) {
    277. 

  277.                             $item = &$dom->create_element($k);
    278. 

  278.                             $item->set_content(utf8_encode($this->entity_decode(strval($v))));
    279. 

  279.                             $res->append_child($item);
    280. 

  280.                         }
    281. 

  281.                          
    282. 

  282.                        $elem->append_child($res);
    283. 

  283.                 }
    284. 

  284.             }
    285. 

  285.         }
    286. 

  286.         
    287. 

  287.         return $dom;
    288. 

  288.     }
