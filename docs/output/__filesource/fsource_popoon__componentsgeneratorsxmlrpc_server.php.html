  28. /**
    29. 

  29. * generator for handling xmlrpc requests
    30. 

  30. *
    31. 

  31. * this generator enables you to act as a xmlrpc server. it's usage is very
    32. 

  32. * simple and straightforward.
    33. 

  33. *
    34. 

  34. * what it does:
    35. 

  35. *   just answer requests coming from xmlrpc clients and map them to class
    36. 

  36. *   methods
    37. 

  37. *
    38. 

  38. * what it needs:
    39. 

  39. *   XML_RPC from pear
    40. 

  40. *
    41. 

  41. * how to implement your own server:
    42. 

  42. *   you need to extend the base xmlrp_server generator class and add all
    43. 

  43. *   the methods you want to expose to the public to the extended class.
    44. 

  44. *   finally you have to tell xmlrpc_server about which rpc methods match
    45. 

  45. *   which class methods. - that's all, your application now speaks xmlrpc.
    46. 

  46. *
    47. 

  47. * here is an example of a very simple xmlrpc generator which implements the
    48. 

  48. * obligatory hello world example :)
    49. 

  49. *   class generator_xmlrpc_server_simple extends generator_xmlrpc_server {
    50. 

  50. *   
    51. 

  51. *       function generator_xmlrpc_server_simple(&$sitemap) {
    52. 

  52. *           parent::generator_xmlrpc_server($sitemap);
    53. 

  53. *           
    54. 

  54. *           $this->addDispatch('moblog.helloWorld', '_helloWorld');
    55. 

  55. *       }
    56. 

  56. *   
    57. 

  57. *       function _helloWorld($params) {
    58. 

  58. *           return new XML_RPC_Value('hello world');
    59. 

  59. *       }
    60. 

  60. *   }
    61. 

  61. *
    62. 

  62. *
    63. 

  63. * @author   Philipp Stucki <philipp@bitflux.ch>
    64. 

  64. * @version  $Id: fsource_popoon__componentsgeneratorsxmlrpc_server.php.html,v 1.2 2004/03/08 07:37:47 chregu Exp $
    65. 

  65. * @package  popoon
    66. 

  66. */

    71. 

  71.     * array containing dispatch map

  72.     * @var array

  73.     * @access private

  74.     */
    75. 

  75.     var $_dispatchMap;

    77. 

  77.     /**
    78. 

  78.     * xmlrpc server object
    79. 

  79.     * @var object 
    80. 

  80.     * @access private
    81. 

  81.     */
    82. 

  82.     var $_server;
    83. 

  83.  
    84. 

  84.     /**
    85. 

  85.     * generator id, used when creating a global self-reference.
    86. 

  86.     * @var string 
    87. 

  87.     * @access private
    88. 

  88.     */
    89. 

  89.     var $_generatorID;
    90. 

  90.     
    91. 

  91.     function generator_xmlrpc_server(&$sitemap) {
    92. 

  92.         // generate generator id
    93. 

  93.         $this->_generatorID = $this->_getGeneratorID();
    94. 

  94.  
    95. 

  95.         $this->generator($sitemap);
    96. 

  96.     }
    97. 

  97.     
    98. 

  98.     function init($attribs) {
    99. 

  99.         // call parent method
    100. 

  100.         parent::init($attribs);
    101. 

  101.  
    102. 

  102.         // create a new xmlrpc server
    103. 

  103.         $this->_server = &new XML_RPC_Server($this->_dispatchMap, FALSE);
    104. 

  104.  
    105. 

  105.         // create a global self-reference for doing callbacks
    106. 

  106.         $GLOBALS['_popoon_generator_xmlrpc_server'][$this->_generatorID] = &$this;
    107. 

  107.     }    
    108. 

  108.  
    109. 

  109.     /**
    110. 

  110.     * create an id for this generator. used when creating a global self-reference.
    111. 

  111.     * @return string generator id
    112. 

  112.     */
    113. 

  113.     function _getGeneratorID() {
    114. 

  114.         return get_class($this);
    115. 

  115.     }
    116. 

  116.     
    117. 

  117.     function DomStart(&$xml) {
    118. 

  118.         // parse request
    119. 

  119.         $r = $this->_server->parseRequest($GLOBALS['HTTP_RAW_POST_DATA']);
    120. 

  120.  
    121. 

  121.         // and serialize the result - that's it.
    122. 

  122.         $xml = '<?xml version="1.0" ?>'.$r->serialize();
    123. 

  123.         $this->sitemap->setHeader('Content-length', strlen($xml));
    124. 

  124.     }
    125. 

  125.     
    126. 

  126.     /**
    127. 

  127.     * adds a method to the dispatch map
    128. 

  128.     * @param string $methodname name of rpc method
    129. 

  129.     * @param string $functionName name of function to call
    130. 

  130.     * @return bool returns true when method has been added to the dispatch map
    131. 

  131.     */
    132. 

  132.     function addDispatch($methodName, $functionName) {
    133. 

  133.         if(method_exists($this, $functionName)) {
    134. 

  134.             $this->_dispatchMap[$methodName] =  array('function' => '$GLOBALS[\'_popoon_generator_xmlrpc_server\'][\''.$this->_generatorID.'\']->'.$functionName);
    135. 

  135.             return TRUE;
    136. 

  136.         }
    137. 

  137.  
    138. 

  138.         return FALSE;
    139. 

  139.     }
    140. 

  140. }
