    71. 

  71.             $uri = $this->db->quote($xmlDoc->dump_node($nodeSet[0]));
    72. 

  72.

  74.             //$uri = $this->db->quote($this->_generateURI($uri));

  77.             $xpRes = $this->_evalXPath($xmlDoc, '/iba/page/Article/lang/text()');

  78.             $nodeSet = $xpRes->nodeset;
    79.

  88.             $nodeSet = $xpRes->nodeset;

    90. 

  90.  
    91. 

  91.             // get article content

  92.             $xpRes = $this->_evalXPath($xmlDoc, '/iba/page/Article/main/child::*');
    93. 

  93.             $nodeSet = $xpRes->nodeset;
    94. 

  94.             foreach($nodeSet as $node) {
    95. 

  95.                 $main .= $xmlDoc->dump_node($node);
    96. 

  96.             }
    97. 

  97.             $main = $this->db->quote($main);
    98. 

  98.  
    99. 

  99.             // check wether this articlee already exists
    100. 

  100.             if(($articleID = $this->_getArticleID($sectionID, $uri)) !== FALSE) {
    101. 

  101.  
    102. 

  102.                 // update existing article
    103. 

  103.                 $this->_updateArticle($articleID, $title, $lang, $main, $author);
    104. 

  104.             
    105. 

  105.             } else {
    106. 

  106.  
    107. 

  107.                 // insert a new section
    108. 

  108.                 $newSectionID = $this->_insertSection($sectionID, $uri, $title, $author);
    109. 

  109.                 
    110. 

  110.                 // insert a new document
    111. 

  111.                 $newDocumentID = $this->_insertDocument($newSectionID, $uri, $title, $author);
    112. 

  112.     
    113. 

  113.                 // insert a new article
    114. 

  114.                 $newArticleID = $this->_insertArticle($newDocumentID, $title, $lang, $author, $main);
    115. 

  115.             }
    116. 

  116.  
    117. 

  117.             // flush output cache
    118. 

  118.             $this->_deleteCache();            
    119. 

  119.         }
    120. 

  120.  
    121. 

  121.         $this->sitemap->setResponseCode(204);
    122. 

  122.         return array("message" => "Data saved");
    123. 

  123.     }

  347.     function _generateURI($title) {

  349.         // replace numeric entities by its characters and vaporise all other lethal characters

    351. 

  351.             '/&#x([\dA-F]+);/e',
    352. 

  352.             '/\n\r\t/'
    353.

    355. 

  355.             "chr(hexdec('\\1'))",
    356. 

  356.             ''

  358.
