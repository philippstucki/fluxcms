  32. /** 
    33. 

  33. * Some words about 304 caching:
    34. 

  34. *
    35. 

  35. * It has now quite sophisticated 304 detecting mechanism built in
    36. 

  36. *
    37. 

  37. * C-Etag = Cached Etag (md5($content))
    38. 

  38. * B-Etag = Browser sent etag
    39. 

  39. * C-LM = Cached Last Modified
    40. 

  40. * B-LM = Browser Last Modified
    41. 

  41. * M-LM = Last Modified in oc.meta
    42. 

  42. *
    43. 

  43. * oc.meta = outputcache.meta. Dies directory contains metadata about LM and Etag of
    44. 

  44. *  the last cached url.
    45. 

  45. *  This allows to delete the cache directory (outputcache) completely, but we still now
    46. 

  46. *   the Etag (md5 of content) and LM of the last generated page.
    47. 

  47. *
    48. 

  48. * If url exits in cache and C-ETag = B-Etag and C-LM <= B-LM
    49. 

  49. *   send 304
    50. 

  50. *
    51. 

  51. * If url does not exist in cache and not in outputcache.meta
    52. 

  52. *   generate content
    53. 

  53. *   generate C-Etag = md5($content)
    54. 

  54. *   if C-Etag = B-Etag
    55. 

  55. *      send 304
    56. 

  56. *   save content
    57. 

  57. *   save oc.meta
    58. 

  58. *   print content
    59. 

  59. *   // this method saves us sending the whole content, when the browser already cached this content
    60. 

  60. *   // it does not make it faster, since we have to generate the content anyway, we just save bandwidth
    61. 

  61. *
    62. 

  62. * If url does not exist in cache but in outputcache.meta
    63. 

  63. *   generate content
    64. 

  64. *   generate C-Etag = md5($content)
    65. 

  65. *   if C-Etag = M-Etag
    66. 

  66. *      set LM to M-LM
    67. 

  67. *   if C-Etag = B-Etag
    68. 

  68. *      send 304
    69. 

  69. *   if C-LM <= B-LM
    70. 

  70. *      send 304
    71. 

  71. *   save content
    72. 

  72. *   print content
    73. 

  73. *
    74. 

  74. *   //this method saves us also sending the whole content as above, it additionally keeps the LM if the
    75. 

  75. *   // md5 did not change. this is more cosmetics than really saving resources as the above does save
    76. 

  76. *   // the resources anyway (except the browser doesn't understand Etags..)
    77. 

  77. *
    78. 

  78. *   Honestly said, the later 2 parts in the 304 handler is maybe overkill... It's just cosmetics more or
    79. 

  79. *    less and will not save much resources. But with newsreaders, which read the rss file every half an
    80. 

  80. *    hour or so, it could sum up and save some bandwidth.
    81. 

  81. * 
    82. 

  82. *  Maybe I will make the later part with the extra oc.meta files optional... But for Google it's certainly
    83. 

  83. *   nice to know, when the page was really last modified and not, when the cache was last created ;)
    84. 

  84. */

  88.     function __construct(popoon_classes_config $options = NULL) {
    89. 

  89.         require_once('Cache/Output.php');
    90. 

  90.         $this->cache = new Cache_Output($options->cacheContainer, $options->cacheParams );
    91. 

  91.     }

  93.     function start() {
    94. 

  94.         $idParams = $_GET;
    95. 

  95.         if (isset($idParams['SID']))
    96. 

  96.         {
    97. 

  97.             unset($idParams['SID']);
    98. 

  98.         }
    99. 

  99.         $this->id = $this->cache->generateID($idParams);
    100. 

  100.         if ($content = $this->cache->start($this->id,'outputcache') ) {
    101. 

  101.             
    102. 

  102.             $header = unserialize($this->cache->getUserdata($this->id,'outputcache'));
    103. 

  103.             $etag = $header['ETag'];
    104. 

  104.             foreach ($header as $key => $value) {
    105. 

  105.                 header("$key: $value");
    106. 

  106.             }
    107. 

  107.             if ($this->check304($etag, $header['Last-Modified'])) {
    108. 

  108.                 header( 'HTTP/1.1 304 Not Modified' );
    109. 

  109.                 die();
    110. 

  110.             }
    111. 

  111.             
    112. 

  112.             header("X-Popoon-Cached: true");
    113. 

  113.             print $content;
    114. 

  114.             die();
    115. 

  115.         }
    116. 

  116.     }
    117.

  118.     function check304($etag, $lastModified) {
    119. 

  119.         if (isset($_SERVER["HTTP_IF_NONE_MATCH"])) {
    120. 

  120.             if ($etag == stripslashes($_SERVER["HTTP_IF_NONE_MATCH"])) {
    121. 

  121.                 return true;
    122. 

  122.             }
    123. 

  123.         }
    124. 

  124.         else if (isset($_SERVER["HTTP_IF_MODIFIED_SINCE"]) )
    125. 

  125.         {
    126. 

  126.             if (strtotime($lastModified) <= strtotime($_SERVER["HTTP_IF_MODIFIED_SINCE"])) {
    127. 

  127.                 return true;
    128. 

  128.             } 
    129. 

  129.         }
    130. 

  130.         return false;
    131. 

  131.     }

  133.     function end(&$sitemap, $expire = 3600) {
    134. 

  134.         $content =  ob_get_contents();
    135. 

  135.         ob_end_clean();
    136. 

  136.         $etag =  md5($content);
    137. 

  137.         $sitemap->setHeader("ETag", $etag);
    138. 

  138.         $metadata = $this->cache->get($this->id.'.meta','outputcache.meta');
    139. 

  139.         $lastModified = null;
    140. 

  140.         if ($metadata) {
    141. 

  141.             if (isset($metadata['Etag'] )&& $metadata['Etag'] == $etag) {
    142. 

  142.                 $sitemap->setHeaderIfNotExists("Last-Modified",$metadata['Last-Modified']);
    143. 

  143.                 $lastModified = true;
    144. 

  144.             } 
    145. 

  145.         }
    146. 

  146.         if (!$lastModified) {
    147. 

  147.             $metadata['Etag'] = $etag;
    148. 

  148.             $sitemap->setHeaderIfNotExists("Last-Modified",gmdate('D, d M Y H:i:s T'));
    149. 

  149.             $metadata['Last-Modified'] = $sitemap->header['Last-Modified'];
    150. 

  150.             $this->cache->container->save($this->id.'.meta', $metadata, 0 ,"outputcache.meta","" );
    151. 

  151.         }
    152. 

  152.         
    153. 

  153.         // we don't want phps cache-control stuff, when we do caching
    154. 

  154.         // there is a "problem" if session_start is used, then PHP adds no-cache http headers
    155. 

  155.         // we do not want that in outputCaching.
    156. 

  156.         // Drawback: OutputCAching with sites relying on different sessions-values do not work
    157. 

  157.         header("Pragma: ");
    158. 

  158.         header("Cache-Control: ");
    159. 

  159.         header("Expires: ");
    160. 

  160.         
    161. 

  161.         foreach ($sitemap->header as $key => $value) {
    162. 

  162.             header("$key: $value");
    163. 

  163.         }
    164. 

  164.         
    165. 

  165.         
    166. 

  166.         if ($this->check304($etag, $sitemap->header['Last-Modified'])) {
    167. 

  167.             header( 'HTTP/1.1 304 Not Modified' );
    168. 

  168.             $this->cache->container->save($this->id, $content, $expire ,"outputcache", serialize($sitemap->header));
    169. 

  169.             die();
    170. 

  170.         } else {
    171. 

  171.             print $content;
    172. 

  172.             $this->cache->container->save($this->id, $content, $expire ,"outputcache", serialize($sitemap->header));
    173. 

  173.         }
    174. 

  174.     }
